				has_ethic = ethic_fanatic_authoritarian
			}
		}
	}
	allow = {
		NOT = {
			has_modifier = patrolling_society

		}
	}
	effect = {
		add_modifier = {
			modifier = "patrolling_society"
		}

			add_deposit = d_patrolling_society_deposit
		}
	}
	ai_weight = {
		weight = 0.25
		modifier = {
			factor = 0
			planet_crime <= 5
		}
		modifier = {
			factor = 8
			planet_crime >= 10
		}
		modifier = {
			factor = 8
			planet_stability <= 70
		}
	}
